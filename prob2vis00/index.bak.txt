  <script>
    window.__timelines = window.__timelines || {};
    const tl = gsap.timeline({ paused: true });
    window.__timelines["main"] = tl;

    const passVal = document.getElementById('pass-val');
    const statusVal = document.getElementById('status-val');
    const pulse = document.getElementById('pulse');

    function resetUI() {
      gsap.set(".logic-card", { className: "logic-card" });
      gsap.set(".flow-line", { stroke: "rgba(255,255,255,0.1)", strokeWidth: 4 });
    }

    function animatePath(pathId, color, startTime, duration = 0.5) {
      const path = document.getElementById(pathId);
      const length = path.getTotalLength();
      const obj = { d: 0 };

      tl.to(obj, {
        d: length,
        duration: duration,
        onUpdate: () => {
          const pt = path.getPointAtLength(obj.d);
          gsap.set(pulse, {
            left: pt.x - 14, // Center pulse
            top: pt.y - 14,
            opacity: 1,
            backgroundColor: color,
            boxShadow: `0 0 20px ${color}`
          });
        },
        onComplete: () => {
          gsap.set(pulse, { opacity: 0 });
          gsap.set(path, { stroke: color, strokeWidth: 6 });
        }
      }, startTime);
    }

    // 0. Intro
    tl.to(".clip", { opacity: 1, stagger: 0.2, duration: 1 }, 0);

    // Case 1: VERY WEAK (12345)
    tl.to({}, { onStart: () => { resetUI(); passVal.textContent = ""; statusVal.textContent = "..."; statusVal.style.color = "white"; }, duration: 0.1 }, 1.5);
    "12345".split('').forEach((char, i) => {
      tl.to({}, { onStart: () => { passVal.textContent += char; }, duration: 0.1 }, 1.6 + (i * 0.1));
    });
    animatePath("p-start", "#ef4444", 2.2);
    animatePath("p-short", "#ef4444", 2.7);
    animatePath("p-to-vweak", "#ef4444", 3.2);
    tl.to("#c-vweak", { className: "logic-card card-vweak active", duration: 0.4 }, 3.5);
    tl.to(statusVal, { text: "VERY WEAK", color: "#ef4444", onStart: () => { statusVal.textContent = "VERY WEAK"; }, duration: 0.1 }, 3.7);

    // Case 2: WEAK (secure)
    tl.to({}, { onStart: () => { resetUI(); passVal.textContent = ""; statusVal.textContent = "..."; statusVal.style.color = "white"; }, duration: 0.1 }, 5);
    "secure".split('').forEach((char, i) => {
      tl.to({}, { onStart: () => { passVal.textContent += char; }, duration: 0.1 }, 5.1 + (i * 0.1));
    });
    animatePath("p-start", "#f97316", 5.7);
    animatePath("p-short", "#f97316", 6.2);
    animatePath("p-to-weak", "#f97316", 6.7);
    tl.to("#c-weak", { className: "logic-card card-weak active", duration: 0.4 }, 7.2);
    tl.to(statusVal, { text: "WEAK", color: "#f97316", onStart: () => { statusVal.textContent = "WEAK"; }, duration: 0.1 }, 7.4);

    // Case 3: STRONG (Secret123)
    tl.to({}, { onStart: () => { resetUI(); passVal.textContent = ""; statusVal.textContent = "..."; statusVal.style.color = "white"; }, duration: 0.1 }, 8.5);
    "Secret123".split('').forEach((char, i) => {
      tl.to({}, { onStart: () => { passVal.textContent += char; }, duration: 0.1 }, 8.6 + (i * 0.1));
    });
    animatePath("p-start", "#38bdf8", 9.5);
    animatePath("p-long", "#38bdf8", 10);
    animatePath("p-to-strong", "#38bdf8", 10.5);
    tl.to("#c-strong", { className: "logic-card card-strong active", duration: 0.4 }, 11);
    tl.to(statusVal, { text: "STRONG", color: "#38bdf8", onStart: () => { statusVal.textContent = "STRONG"; }, duration: 0.1 }, 11.2);

    // Case 4: VERY STRONG (V3ry$ecure!)
    tl.to({}, { onStart: () => { resetUI(); passVal.textContent = ""; statusVal.textContent = "..."; statusVal.style.color = "white"; }, duration: 0.1 }, 12);
    "V3ry$ecure!".split('').forEach((char, i) => {
      tl.to({}, { onStart: () => { passVal.textContent += char; }, duration: 0.08 }, 12.1 + (i * 0.08));
    });
    animatePath("p-start", "#10b981", 12.8);
    animatePath("p-long", "#10b981", 13.3);
    animatePath("p-to-vstrong", "#10b981", 13.8);
    tl.to("#c-vstrong", { className: "logic-card card-vstrong active", duration: 0.4 }, 14.3);
    tl.to(statusVal, { text: "VERY STRONG", color: "#10b981", onStart: () => { statusVal.textContent = "VERY STRONG"; }, duration: 0.1 }, 14.5);

    // Outro
    tl.to("#root", { opacity: 0, duration: 0.4 }, 14.8);

  </script>
